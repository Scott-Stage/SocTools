Import-Module ActiveDirectory

$GUID_ENROLL = [guid]'0e10c968-78fb-11d2-90d4-00c04f79dc55'

$cfg  = (Get-ADRootDSE).configurationNamingContext
$base = "CN=Enrollment Services,CN=Public Key Services,CN=Services,$cfg"

$cas = Get-ADObject -SearchBase $base -LDAPFilter '(objectClass=pKIEnrollmentService)' `
                    -Properties cn

$caPerms = foreach ($ca in $cas) {
  $acl = Get-Acl -Path ("AD:\" + $ca.DistinguishedName)
  $acl.Access | Where-Object {
      $_.AccessControlType -eq 'Allow' -and
      $_.ActiveDirectoryRights.HasFlag([System.DirectoryServices.ActiveDirectoryRights]::ExtendedRight) -and
      $_.ObjectType -eq $GUID_ENROLL
  } | Select-Object @{n='CACommonName';e={$ca.cn}},
                    @{n='Right';e={'Enroll'}},
                    @{n='Principal';e={$_.IdentityReference.Value}},
                    @{n='IsInherited';e={$_.IsInherited}}
}

$caPerms | Sort-Object CACommonName,Principal | Format-Table -AutoSize
