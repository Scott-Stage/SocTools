private static object? SafeChangeType(object? value, Type targetType)
{
    if (value == null) return null;

    // 1. Unwrap Nullable Types (e.g. int? -> int)
    Type? underlyingType = Nullable.GetUnderlyingType(targetType);
    if (underlyingType != null) targetType = underlyingType;

    // 2. Direct Assignment (e.g. byte[] -> byte[])
    if (targetType.IsInstanceOfType(value)) return value;

    // 3. Handle Binary Data (The source of your errors)
    if (value is byte[] bytes)
    {
        // A. Target is Guid (e.g. public Guid ObjectGuid { get; set; })
        if (targetType == typeof(Guid)) return new Guid(bytes);

        // B. Target is String (e.g. public string ObjectGuid { get; set; })
        if (targetType == typeof(string))
        {
            // Is it a GUID? (16 bytes)
            if (bytes.Length == 16) return new Guid(bytes).ToString();

            // Is it a SID? (Starts with S-...)
            try 
            { 
                 return new System.Security.Principal.SecurityIdentifier(bytes, 0).ToString(); 
            } 
            catch { /* Not a SID, continue */ }

            // Default: Convert Image/Hash to Base64 String so it doesn't crash
            return Convert.ToBase64String(bytes);
        }
    }

    // 4. Fallback for everything else
    return Convert.ChangeType(value, targetType);
}
