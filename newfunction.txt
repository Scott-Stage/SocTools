    <!-- The row that controls it -->
    <MudListItem OnClick="@(()=>Toggle(item.Id))">

@code {
    private HashSet<string> open = new();
    bool IsOpen(string id) => open.Contains(id);
    void Toggle(string id) { if (!open.Add(id)) open.Remove(id); }
}
