public sealed class TenableScResponse<T>
{
    [JsonPropertyName("type")]      public string? Type { get; set; }      // e.g., "regular"
    [JsonPropertyName("response")]  public T? Response { get; set; }       // The actual data
    [JsonPropertyName("error_code")]public int? ErrorCode { get; set; }
    [JsonPropertyName("error_msg")] public string? ErrorMsg { get; set; }
    [JsonPropertyName("timestamp")] public string? Timestamp { get; set; }
}
