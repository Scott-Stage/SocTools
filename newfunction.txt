@using System.Globalization

<div class="drive-bar-container">
    <div class="drive-bar-details">
        <!-- Display the formatted text -->
        <span class="drive-bar-label">@($"{FreeSpaceGB:N1} GB free of {TotalCapacityGB:N1} GB")</span>
    </div>
    
    <!-- The outer bar -->
    <div class="drive-bar-background">
        <!-- The inner bar that shows the used space. Its style is set dynamically. -->
        <div class="drive-bar-used @(IsLowOnSpace ? "low-space" : "")" style="width: @(UsedPercentage.ToString(CultureInfo.InvariantCulture))%;">
        </div>
    </div>
</div>

@code {
    [Parameter]
    public double TotalCapacityGB { get; set; }

    [Parameter]
    public double FreeSpaceGB { get; set; }

    private double UsedSpaceGB => TotalCapacityGB - FreeSpaceGB;
    private double UsedPercentage => (TotalCapacityGB > 0) ? (UsedSpaceGB / TotalCapacityGB * 100) : 0;
    private bool IsLowOnSpace => (TotalCapacityGB > 0) && ((FreeSpaceGB / TotalCapacityGB) < 0.10);

    // This ensures parameters are validated when the component is used.
    protected override void OnInitialized()
    {
        if (TotalCapacityGB < 0 || FreeSpaceGB < 0 || FreeSpaceGB > TotalCapacityGB)
        {
            throw new ArgumentException("Invalid drive capacity or free space values provided.");
        }
    }
}
