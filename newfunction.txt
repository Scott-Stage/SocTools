// 1. Use 'Where' to find ALL matches
// Adding .ToList() executes the query immediately so you can inspect it
var matches = result?.Response?.Usable?
    .Where(s => s.Name == "Database Confidential Zone 72-73")
    .ToList();

// 2. Loop through them to see the IDs
if (matches != null && matches.Count > 0)
{
    Console.WriteLine($"Found {matches.Count} matches:");
    foreach (var scan in matches)
    {
        Console.WriteLine($" - Name: {scan.Name}, ID: {scan.Id}");
    }
}
else
{
    Console.WriteLine("No matches found.");
}
