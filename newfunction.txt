Css
body, html {
    height: 100%;
    margin: 0;
    display: flex;
    justify-content: center;
    align-items: center;
    background-color: #222;
}

.image-container {
    position: relative;
    width: 80%;
    max-width: 800px;
    height: 60vh;
    overflow: hidden;
    border-radius: 10px;
    box-shadow: 0 10px 20px rgba(0,0,0,0.3);
}

.blending-image {
    position: absolute;
    top: 0;
    left: 0;
    width: 100%;
    height: 100%;
    object-fit: cover;
    opacity: 0;
    animation: blend 25s infinite;
}

/* Staggering the animation delay for each image */
.blending-image:nth-child(1) {
    animation-delay: 0s;
}

.blending-image:nth-child(2) {
    animation-delay: 5s;
}

.blending-image:nth-child(3) {
    animation-delay: 10s;
}

.blending-image:nth-child(4) {
    animation-delay: 15s;
}

.blending-image:nth-child(5) {
    animation-delay: 20s;
}

@keyframes blend {
    0% {
        opacity: 0;
    }
    20% {
        opacity: 1;
    }
    40% {
        opacity: 0;
    }
    100% {
        opacity: 0;
    }
}
