function findObjectWithValue(obj, targetValue) {
    for (const key in obj) {
        if (obj.hasOwnProperty(key)) {
            const value = obj[key];

            if (value === targetValue) {
                return obj;  // Found the value, return the object it's in
            }

            if (typeof value === 'object' && value !== null) {
                const result = findObjectWithValue(value, targetValue);
                if (result) {
                    return result; // Value found in nested object, return the object
                }
            }
        }
    }
    return null; // Value not found
}
