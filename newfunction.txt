private async Task OnItemDropped(MudItemDropInfo<DashboardItem> info)
{
    var item = info.Item;  // The thing that got dragged
    var newIndex = info.Index; // The position it was dropped into

    // Remove old position
    dashboardItems.Remove(item);

    // Insert at new position (safely)
    dashboardItems.Insert(
        Math.Min(newIndex, dashboardItems.Count),
        item
    );

    await InvokeAsync(StateHasChanged);
}
