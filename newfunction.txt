# XML File Path
$xmlFilePath = "C:\path\to\your\your_file.xml"  # Replace with the actual path to your XML file

try {
    # Read XML file and convert to XML object
    $xmlString = Get-Content -Path $xmlFilePath -Raw
    $xmlDocument = [xml]$xmlString

    # Get the root element
    $rootElement = $xmlDocument.report #Adjusted to access the 'report' root.  Change if needed
    # Convert the XML to a hashtable
    $dataHashtable = Convert-XmlElementToHashtable -Element $rootElement

    # Convert hashtable to JSON
    $jsonData = $dataHashtable | ConvertTo-Json -Depth 10

    # Output the JSON data
    Write-Host $jsonData
}
catch {
    Write-Error "Error reading or processing XML file: $($_.Exception.Message)"
}
