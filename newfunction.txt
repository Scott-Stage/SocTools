public sealed class TenableScanListResult<T>
{
    // The list of scans the user has permission to launch/view
    [JsonPropertyName("usable")] 
    public List<T>? Usable { get; set; }

    // The list of scans the user has permission to edit/configure
    [JsonPropertyName("manageable")] 
    public List<T>? Manageable { get; set; }
}
