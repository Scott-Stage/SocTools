import time
import sys

def simple_progress_bar(iteration, total, bar_length=50):
    """
    Displays a simple progress bar.
    
    Args:
        iteration (int): The current iteration number (should be 0-indexed).
        total (int): The total number of iterations.
        bar_length (int): The character length of the progress bar.
    """
    # Calculate the percentage complete
    percent = 100.0 * (iteration + 1) / total
    
    # Calculate the number of filled blocks in the bar
    filled_length = int(bar_length * (iteration + 1) // total)
    
    # Create the bar string
    bar = 'â–ˆ' * filled_length + '-' * (bar_length - filled_length)
    
    # The final output string
    # \r moves the cursor to the beginning of the line
    # The end=' ' prevents the default newline character
    output_str = f'\rProgress: |{bar}| {percent:.1f}% Complete'
    
    # Write the output and flush it to the screen
    sys.stdout.write(output_str)
    sys.stdout.flush()

# --- Example Usage ---
print("Starting a simple task...")
total_items = 200
for i in range(total_items):
    # Do some work
    time.sleep(0.02)
    # Update the progress bar
    simple_progress_bar(i, total_items)

# Print a newline at the end to move to the next line
print("\nTask finished!")
