    // 1. Map the raw API field (String)
    [JsonPropertyName("scanDuration")]
    public string? ScanDurationString { get; set; }

    // 2. Helper property to get actual TimeSpan
    [JsonIgnore]
    public TimeSpan Duration
    {
        get
        {
            if (long.TryParse(ScanDurationString, out long seconds))
            {
                return TimeSpan.FromSeconds(seconds);
            }
            return TimeSpan.Zero;
        }
    }

    // 3. Helper to get pretty text (e.g., "00:02:34")
    [JsonIgnore]
    public string DurationFormatted => Duration.ToString(@"hh\:mm\:ss");
