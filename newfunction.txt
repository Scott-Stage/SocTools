    const parser = new DOMParser();
    const xmlDoc = parser.parseFromString(xmlString, "text/xml");

    // Handle XML parse errors (check for <parsererror> tag)
    if (xmlDoc.getElementsByTagName("parsererror").length > 0) {
      console.error("XML Parsing Error:", xmlDoc.getElementsByTagName("parsererror")[0].textContent);
      return; // Stop processing this event if there's an error
    }

    // Now you can work with the xmlDoc (the DOM tree)
    // Example: Extracting a title from the XML:
    const titleElement = xmlDoc.getElementsByTagName("title")[0];
    if (titleElement) {
      const title = titleElement.textContent;
      console.log("Title:", title);
